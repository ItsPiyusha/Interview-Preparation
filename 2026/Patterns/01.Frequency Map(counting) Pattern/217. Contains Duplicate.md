## Fix 1: minimal changes to your approach (2-pass)
```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_map<int,int> cnt;
        for (int x : nums) cnt[x]++;

        for (auto &p : cnt) {          // p is (key, value)
            if (p.second > 1) return true;
        }
        return false;
    }
};
```
* ```p.first``` is the number
* ```p.second``` is its frequency

## Fix 2: best interview version (1-pass early return)
This is the usual clean pattern:
```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_set<int> seen;
        for (int x : nums) {
            if (seen.count(x)) return true;
            seen.insert(x);
        }
        return false;
    }
};
```
## If you want to keep unordered_map but still 1-pass
```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_map<int,int> cnt;
        for (int x : nums) {
            if (++cnt[x] > 1) return true; //“After counting this occurrence of x, is its frequency at least 2?”
        }
        return false;
    }
};
```
